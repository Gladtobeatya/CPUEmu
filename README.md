# CPUEmu
 CPU Emulator

Existing regs : R0 to R9
Avaliable ops : 
 SET REG REG/IMM -> set 
 ADD REG REG/IMM -> add
	SUB REG REG/IMM -> substract
	JMP IMM         -> jump to
	PRINT REG/IMM   -> print
	MUL REG REG/IMM -> multiply
	DIV REG REG/IMM -> divide
	INC REG         -> increment
	DEC REG         -> decrement
	CMP REG REG/IMM -> compare
	JZ IMM          -> jump if flag TRUE
	JNZ IMM         -> jump if flag FALSE
	MOV REG REG/IMM -> copy
	HALT            -> stop CPU

 Edit input.txt to simulate!
